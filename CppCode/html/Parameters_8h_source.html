616 template <int d>
617 void Parameters<d>::init_locations (char *line, v2d & X)
618 {
619  if (!strcmp(line, "square"))
620  {
621  auto m = *(std::max_element(r.begin(), r.end())) ;
622  printf("%g\n", m) ;
623  int dd ;
624  for (dd=0 ; dd<d ; dd++) X[0][dd]=Boundaries[dd][0]+m ;
625  for (int i=1 ; i<N ; i++)
626  {
627  X[i]=X[i-1] ;
628  for (dd=0 ; dd<d ; dd++)
629  {
630  X[i][dd] += 2*m ;
631  if (X[i][dd]>Boundaries[dd][1]-m)
632  X[i][dd] = Boundaries[dd][0]+m ;
633  else
634  break ;
635  }
636  if (dd==d) {printf("WARN: cannot affect all particles on the square lattice, not enough space in the simulation box\n") ; break ; }
637  }
638  }
639  else if (!strcmp(line, "randomdrop"))
640  {
641  boost::random::mt19937 rng(seed);
642  boost::random::uniform_01<boost::mt19937> rand(rng) ;
643 
644  for (int i=0 ; i<N ; i++)
645  {
646  for(int dd=0 ; dd < d ; dd++)
647  {
648  if (Boundaries[dd][3]==0)
649  X[i][dd] = rand()*Boundaries[dd][2] + Boundaries[dd][0] ;
650  else
651  X[i][dd] = rand()*(Boundaries[dd][2]-2*r[i]) + Boundaries[dd][0] + r[i] ;
652  }
653  }
654 
655  }
656  else if (!strcmp(line, "roughinclineplane"))
657  {
658  boost::random::mt19937 rng(seed);
659  boost::random::uniform_01<boost::mt19937> rand(rng) ;
660  printf("Location::roughinclineplane assumes a plane of normal [1,0,0...] at location 0 along the 1st dimension.") ; fflush(stdout) ;
661  auto m = *(std::max_element(r.begin(), r.end())) ; // Max radius
662  double delta=0.1*m ;
663  for (int dd=0 ; dd<d ; dd++) X[0][dd]=Boundaries[dd][0]+m+delta ;
664  Frozen[0]=true ;
665  for (int i=1 ; i<N ; i++)
666  {
667  X[i]=X[i-1] ;
668  for (int dd=d-1 ; dd>=0 ; dd--)
669  {
670  X[i][dd] += 2*m+2*delta ;
671  if (X[i][dd]>Boundaries[dd][1]-m-delta)
672  X[i][dd] = Boundaries[dd][0]+m+delta ;
673  else
674  break ;
675  }
676  if (X[i][0]==Boundaries[0][0]+m+delta) Frozen[i]=true ;
677  // randomize the previous grain
678  for (int dd=0 ; dd<d ; dd++)
679  X[i-1][dd] += (rand()-0.5)*2*delta ;
680  }
681  }
682  else if (!strcmp(line, "roughinclineplane2"))
683  {
684  boost::random::mt19937 rng(seed);
685  boost::random::uniform_01<boost::mt19937> rand(rng) ;
686  printf("Location::roughinclineplane assumes a plane of normal [1,0,0...] at location 0 along the 1st dimension.") ; fflush(stdout) ;
687  auto m = *(std::max_element(r.begin(), r.end())) ; // Max radius
688  double delta=0.1*m ; int ddd ;
689  for (int dd=0 ; dd<d ; dd++) X[0][dd]=Boundaries[dd][0]+m ;
690  Frozen[0]=true ; bool bottomlayer=true ;
691  for (int i=1 ; i<N ; i++)
692  {
693  X[i]=X[i-1] ;
694  for (ddd=d-1 ; ddd>=0 ; ddd--)
695  {
696  if (bottomlayer)
697  {
698  X[i][ddd]+= 2*m ;
699  if (X[i][ddd]>Boundaries[ddd][1]-m)
700  X[i][ddd] = Boundaries[ddd][0]+m ;
701  else
702  break ;
703  }
704  else
705  {
706  X[i][ddd] += 2*m+2*delta ;
707  if (X[i][ddd]>Boundaries[ddd][1]-m-delta)
708  X[i][ddd] = Boundaries[ddd][0]+m+delta ;
709  else
710  break ;
711  }
712  }
713  if (ddd==0) bottomlayer=false ;
714  if (bottomlayer)
715  {
716  Frozen[i-1]=true ;
717  X[i-1][0] += rand()*delta ; // Randomness only on x0, and only >r (above the bottom wall)
718  }
719  else
720  {
721  for (int dd=0 ; dd<d ; dd++)
722  X[i-1][dd] += (rand()-0.5)*2*delta ;
723  }
724  }
725  }
726  else if (!strcmp(line, "quasicrystal"))
727  {
728  auto m = *(std::max_element(r.begin(), r.end())) ; // largest radius
729  printf("%g\n", m) ;
730  int dd ;
731  // X is an array of particle locations
732  for (dd=0 ; dd<d ; dd++) { X[0][dd]=Boundaries[dd][0]+m ; } // d is number of dimensions, dd is its index
733  for (int i=1 ; i<N ; i++) // number of particles
734  {
735  X[i]=X[i-1] ; // get previous particle location
736  for (dd=0 ; dd<d ; dd++) // iterate over dimensions
737  {
738  X[i][dd] += 2*m ; // add a diameter
739  if (X[i][dd]>Boundaries[dd][1]-m) // if too close to 'right'
740  X[i][dd] = Boundaries[dd][0]+m ; // bring back to 'left'
741  else
742  break ;
743  }
744  if (dd==d) {printf("WARN: cannot affect all particles on the square lattice, not enough space in the simulation box\n") ; break ; }
745  }
746  }
747  else
748  printf("ERR: undefined initial location function.\n") ;
749 }
