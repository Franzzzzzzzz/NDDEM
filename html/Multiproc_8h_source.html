104 void Multiproc<d>::split(int N, int P)
105 {
106 double Delta,b1,b2 ;
107 int Nprime=N ;
108 
109 double Area=N*(N-1)/(2*P) ;
110 
111 for (int i=0 ; i<P ; i++)
112 {
113  Delta=Nprime*Nprime-2*Area ;
114  if (Delta<0) {printf("Error: Quadratic equation has no solution(multiproc)\n") ; fflush(stdout) ;}
115  b1=(Nprime-sqrt(Delta)) ;
116  b2=(Nprime+sqrt(Delta)) ;
117  if (b1>=0 && b1<N)
118  {
119  printf("%d Using b1 %d %d\n", i, int(floor(b1)), int(floor(b2))) ;
120  share[i+1]=share[i]+int(floor(b1)) ;
121  Nprime -= int(floor(b1)) ;
122  }
123  else if (b2>=0 && b2<N)
124  {
125  printf("%d Using b2 %d\n", i, int(floor(b2))) ;
126  share[i+1]=share[i]+int(floor(b2)) ;
127  Nprime -= int(floor(b2)) ;
128  }
129  else printf("No solution ...") ;
130 }
131 
132 share[P]=N ;
133 }
